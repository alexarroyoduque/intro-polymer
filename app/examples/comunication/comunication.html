Comunicar elementos

Al crear un elemento de Polymer creamos funciones y variables. Buscando ese elemento el DOM es posible ejecutar sus métodos.

Ejemplo

En este ejemplo elementos no están anidados pero también funcionaría si lo estuvieran.

intro-batmovil.html

<dom-module id="intro-batmovil">

  <template>
    <p><span class="title">Batmovil</span></p>
    <p>Misiles disparados: <span>{{shoots}}</span></p>
  </template>

  <style>
    .title {
      background-color: black;
      color: white;
      border: 2px solid gray;
      padding: 5px;
      border-radius: 50%;
    }
  </style>

  <script>
    Polymer({
      is: "intro-batmovil",
      properties: {
        shoots: {
          type: Number,
          value: 0
        }
      },
      shoot: function () {
          this.shoots++;
      }
    });
  </script>

</dom-module>

intro-batman.html

<dom-module id="intro-batman">
  <template>
    <p><span class="title">Batman</span></p>
    <p><button on-click="sendSignal">Disparar misil del batmovil</button></p>
  </template>

  <style>
    .title {
      background-color: black;
      color: white;
      border: 2px solid gray;
      padding: 5px;
    }
  </style>

  <script>
    Polymer({
      is: "intro-batman",
      ready: function () {
        document.addEventListener('WebComponentsReady', (function() {
            this.introBatmovil = document.querySelector('intro-batmovil');
        }).bind(this));
      },
      sendSignal: function () {
        this.introBatmovil.shoot();
      }
    });
  </script>

</dom-module>
