Creando variables

Las variables de un componente siempre han de estar dentro de una etiqueta HTML.

Tipos

  • String
  • Date
  • Number
  • Boolean
  • Array
  • Object

intro-hero.html

<dom-module id="intro-hero">
  <template>
    <p>El hombre del ma√±ana:
      <b>{{name}}</b> <b>{{lastname}}</b>
    </p>
    <p>Vueltas al mundo: <span>{{laps}}</span></p>
    <button on-click="turnTheWorld">Dar vuelta al mundo</button>
  </template>

  <script>
    Polymer({
      is: "intro-hero",
      properties: {
        name: {
          type: String,
          value: 'Clark'
        },
        lastname: {
          type: String,
          value: 'Kent'
        },
        laps: {
          type: Number,
          value: 0
        }
      },
      turnTheWorld: function () {
        this.laps++;
      }
    });
  </script>
</dom-module>
