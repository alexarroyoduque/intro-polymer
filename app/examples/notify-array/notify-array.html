Notificar cambios en Array

Polymer no observa estructuras de datos complejas igual que otros tipos simples como strings, numbers, etc, ...

Polymer proporciona sus propios m√©todos para modificar Arrays:

  • push(path, item1, [..., itemN])
  • pop(path)
  • unshift(path, item1, [..., itemN])
  • shift(path)
  • splice(path, index, removeCount, [item1, ..., itemN])

intro-batman.html

<dom-module id="intro-batman">
  <template>
    <p>Batman</p>
    <p>Batarangs disponibles: <span>{{batarangs.length}}</span></p>
    <button on-click="buyBatarang">Comprar batarang</button>
    <p>Estado del primer Batarang: <span>{{batarangs.0.status}}</span></p>
    <button on-click="useFirstBatarang">Lanzar el primer batarang</button>
  </template>

  <script>
    Polymer({
      is: "intro-batman",
      properties: {
        batarangs: {
          type: Array,
          value: function() {
            return [{status: 'ok'}];
          }
        }
      },
      buyBatarang: function () {
        this.push('batarangs', {status: 'ok'});
      },
      useFirstBatarang: function () {
        this.notifyPath('batarangs.' + 0 + '.status', 'broken');
      }
    });
  </script>
</dom-module>
